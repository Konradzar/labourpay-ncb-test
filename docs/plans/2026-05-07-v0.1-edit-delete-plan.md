```typescript
// === Server-side authenticated NCB fetch ===
// Wraps the Bearer-auth pattern used by Server Components and Server Actions
// when accessing privileged NCB endpoints (single-record reads, updates,
// deletes). The secret key NEVER reaches the browser bundle — this helper is
// safe to import only from server-only modules (Server Components, Server
// Actions, Route Handlers).
//
// Pattern established in commit 1c91eb4 (worker detail page Bearer-auth fix).
export async function ncbAuthFetch(
  pathSuffix: string,
  init: Omit<RequestInit, "headers"> & { headers?: Record<string, string> } = {}
): Promise<Response> {
  const secret = process.env.NCB_SECRET_KEY;
  if (!secret) {
    throw new Error(
      "NCB_SECRET_KEY missing — required for server-side authenticated NCB calls. " +
        "Check .env.local."
    );
  }

  const url = `${CONFIG.dataApiUrl}${pathSuffix}${
    pathSuffix.includes("?") ? "&" : "?"
  }Instance=${CONFIG.instance}`;

  return fetch(url, {
    cache: "no-store",
    ...init,
    headers: {
      Authorization: `Bearer ${secret}`,
      "Content-Type": "application/json",
      "X-Database-Instance": CONFIG.instance,
      ...(init.headers ?? {}),
    },
  });
}
```

```typescript
async function fetchWorker(id: string): Promise<Worker | null> {
  // We use NCB's single-record endpoint /read/workers/<id> with the
  // NCB_SECRET_KEY as a Bearer token (server-side only — never sent to the
  // browser). The previous "list with ?id=<id> filter" workaround returned
  // 500 because NCB's list endpoint only accepts filters on the columns
  // {name, monthly_salary, photo_key, id_doc_key, user_id} — the `id`
  // primary key is NOT a filterable query param.
  const res = await ncbAuthFetch(`/read/workers/${encodeURIComponent(id)}`);
  if (res.status === 404) return null;
  if (!res.ok) {
    throw new Error(
      `NCB get worker failed: ${res.status} ${await res.text()}`
    );
  }
  const json = (await res.json()) as NCBSingleResponse<Worker> | { data?: Worker[] };

  // NCB may return the row as `data: <object>` or `data: [<object>]`.
  // Handle both shapes defensively.
  const data = (json as { data?: unknown }).data;
  if (!data) return null;
  if (Array.isArray(data)) {
    return (data[0] as Worker) ?? null;
  }
  return data as Worker;
}
```

```tsx
"use client";

// app/workers/[id]/edit/EditWorkerForm.tsx
//
// Edit form. Mirrors AddWorkerForm.tsx exactly except:
//   - name + monthly_salary inputs are pre-filled from props.worker
//   - photo / id_doc UploadState starts with the existing key (not null)
//   - on submit, calls updateWorker(id, fields) Server Action (not /create)
//
// File handling is upload-on-pick (same as AddWorkerForm): when the user
// selects a file, it uploads to S3 immediately. State stores the resulting
// key. If the user does NOT pick a new file, the state retains the existing
// key. On submit, whatever key is in state goes to NCB.

import { useState, useRef, FormEvent } from "react";
import type { Worker } from "@/lib/types";
import { updateWorker } from "../actions";

const ALLOWED_TYPES = ["image/jpeg", "image/png", "application/pdf"] as const;
const MAX_BYTES = 5 * 1024 * 1024;

type UploadState = {
  key: string | null;
  uploading: boolean;
  error: string | null;
};

// Same upload helper as AddWorkerForm — could DRY into a shared module
// later, but two copies is fine until we have a third caller.
async function uploadFile(file: File): Promise<string> {
  if (!(ALLOWED_TYPES as readonly string[]).includes(file.type)) {
    throw new Error(
      `Unsupported type: ${file.type || "(unknown)"}. Use JPEG, PNG, or PDF.`
    );
  }
  if (file.size > MAX_BYTES) {
    throw new Error(`File too large (max ${MAX_BYTES / 1024 / 1024} MB).`);
  }

  const presignRes = await fetch("/api/upload-url", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({ contentType: file.type }),
  });
  if (!presignRes.ok) {
    throw new Error(
      `Presign failed: ${presignRes.status} ${await presignRes.text()}`
    );
  }
  const { url, key } = (await presignRes.json()) as {
    url: string;
    key: string;
  };

  const putRes = await fetch(url, {
    method: "PUT",
    headers: { "Content-Type": file.type },
    body: file,
  });
  if (!putRes.ok) {
    throw new Error(
      `S3 upload failed: ${putRes.status} ${putRes.statusText}`
    );
  }

  return key;
}

// Treat empty string and the literal "placeholder"-looking strings as null —
// matches the detail page's isUsableKey() helper.
function displayKey(key: string | null): string {
  if (!key) return "none yet";
  if (key === "placeholder") return "none yet";
  if (key.startsWith("placeholder.")) return "none yet";
  return key;
}

export function EditWorkerForm({ worker }: { worker: Worker }) {
  const [photo, setPhoto] = useState<UploadState>({
    key: worker.photo_key,
    uploading: false,
    error: null,
  });
  const [idDoc, setIdDoc] = useState<UploadState>({
    key: worker.id_doc_key,
    uploading: false,
    error: null,
  });
  const [submitting, setSubmitting] = useState(false);
  const [error, setError] = useState<string | null>(null);

  const photoReqId = useRef(0);
  const idDocReqId = useRef(0);

  const handleFileChange = (
    field: "photo" | "id_doc",
    setState: typeof setPhoto,
    reqIdRef: typeof photoReqId
  ) => async (e: React.ChangeEvent<HTMLInputElement>) => {
    const file = e.target.files?.[0];
    if (!file) return;

    const myReqId = ++reqIdRef.current;
    setState((prev) => ({ ...prev, uploading: true, error: null }));

    try {
      const key = await uploadFile(file);
      // Race guard: only commit if this is still the latest request.
      if (reqIdRef.current === myReqId) {
        setState({ key, uploading: false, error: null });
      }
    } catch (err) {
      if (reqIdRef.current === myReqId) {
        setState((prev) => ({
          ...prev,
          uploading: false,
          error: err instanceof Error ? err.message : String(err),
        }));
      }
    }
  };

  const handleSubmit = async (e: FormEvent<HTMLFormElement>) => {
    e.preventDefault();
    setError(null);

    const formData = new FormData(e.currentTarget);
    const name = String(formData.get("name") ?? "").trim();
    const monthly_salary = Number(formData.get("monthly_salary") ?? 0);

    if (!name) {
      setError("Name is required.");
      return;
    }
    if (!Number.isFinite(monthly_salary) || monthly_salary <= 0) {
      setError("Monthly salary must be a positive number.");
      return;
    }
    if (photo.uploading || idDoc.uploading) {
      setError("Wait for file upload to finish.");
      return;
    }
    if (!photo.key || !idDoc.key) {
      // Shouldn't happen — the worker already has both keys from creation.
      // Defensive guard.
      setError("Photo and ID document are required.");
      return;
    }

    setSubmitting(true);
    try {
      await updateWorker(worker.id, {
        name,
        monthly_salary,
        photo_key: photo.key,
        id_doc_key: idDoc.key,
      });
      // Server Action redirects on success — control doesn't return here.
    } catch (err) {
      // Next.js's redirect() throws a special error that's not really an error —
      // re-throw to let Next.js handle the navigation. Anything else is a real
      // error to display.
      if (
        err &&
        typeof err === "object" &&
        "digest" in err &&
        typeof (err as { digest: unknown }).digest === "string" &&
        ((err as { digest: string }).digest as string).startsWith(
          "NEXT_REDIRECT"
        )
      ) {
        throw err;
      }
      setError(err instanceof Error ? err.message : String(err));
      setSubmitting(false);
    }
  };

  const labelStyle: React.CSSProperties = {
    display: "block",
    marginTop: "1.25rem",
    fontWeight: 600,
  };
  const inputStyle: React.CSSProperties = {
    display: "block",
    width: "100%",
    padding: "0.5rem",
    marginTop: "0.25rem",
    fontSize: "1rem",
    fontFamily: "inherit",
    border: "1px solid #ccc",
    borderRadius: 4,
    boxSizing: "border-box",
  };
  const errorStyle: React.CSSProperties = {
    color: "#d63031",
    fontSize: "0.85rem",
    marginTop: "0.25rem",
  };
  const helpStyle: React.CSSProperties = {
    color: "#666",
    fontSize: "0.85rem",
    marginTop: "0.25rem",
  };

  return (
    <form onSubmit={handleSubmit}>
      <label style={labelStyle}>
        Name *
        <input
          type="text"
          name="name"
          defaultValue={worker.name}
          required
          style={inputStyle}
        />
      </label>

      <label style={labelStyle}>
        Monthly salary (Rand) *
        <input
          type="number"
          name="monthly_salary"
          defaultValue={worker.monthly_salary}
          min="1"
          step="1"
          required
          style={inputStyle}
        />
      </label>

      <label style={labelStyle}>
        Photo
        <p style={helpStyle}>
          Current: {displayKey(photo.key)}
        </p>
        <input
          type="file"
          accept="image/jpeg,image/png,application/pdf"
          onChange={handleFileChange("photo", setPhoto, photoReqId)}
          style={{ marginTop: "0.5rem" }}
        />
        <p style={helpStyle}>
          Pick a new file to replace it, or leave blank to keep current.
        </p>
        {photo.uploading && <p style={helpStyle}>Uploading…</p>}
        {photo.error && <p style={errorStyle}>{photo.error}</p>}
      </label>

      <label style={labelStyle}>
        ID Document
        <p style={helpStyle}>
          Current: {displayKey(idDoc.key)}
        </p>
        <input
          type="file"
          accept="image/jpeg,image/png,application/pdf"
          onChange={handleFileChange("id_doc", setIdDoc, idDocReqId)}
          style={{ marginTop: "0.5rem" }}
        />
        <p style={helpStyle}>
          Pick a new file to replace it, or leave blank to keep current.
        </p>
        {idDoc.uploading && <p style={helpStyle}>Uploading…</p>}
        {idDoc.error && <p style={errorStyle}>{idDoc.error}</p>}
      </label>

      {error && (
        <p style={{ ...errorStyle, marginTop: "1rem" }}>
          {error}
        </p>
      )}

      <div style={{ marginTop: "2rem", display: "flex", gap: "1rem" }}>
        <button
          type="submit"
          disabled={submitting || photo.uploading || idDoc.uploading}
          style={{
            background: "#1a73e8",
            color: "white",
            border: "none",
            padding: "0.5rem 1.5rem",
            borderRadius: 4,
            cursor: submitting ? "not-allowed" : "pointer",
            opacity: submitting ? 0.6 : 1,
            fontFamily: "inherit",
            fontSize: "0.95rem",
          }}
        >
          {submitting ? "Saving…" : "Save changes"}
        </button>
        <a
          href={`/workers/${worker.id}`}
          style={{
            color: "#666",
            textDecoration: "none",
            padding: "0.5rem 1.5rem",
          }}
        >
          Cancel
        </a>
      </div>
    </form>
  );
}
```
